create table STUDENT (
	STU_NUM INT(10) PRIMARY KEY, 
    STU_NAME VARCHAR(100),
    STU_GRADE INT(4),
    STU_EMAIL VARCHAR(200),
    STU_GENDER VARCHAR(2),
    STU_PHONE INT(11),
    STU_AGE INT(3),
    STU_ADDRESS VARCHAR(100),
    STU_ID VARCHAR(100),
    STU_PW VARCHAR(100),
    STU_IMG VARCHAR(200)
);


CREATE TABLE JOB_FIELD (
	COMP_ID INT(10) PRIMARY KEY,
    IT VARCHAR(10),
    SALES VARCHAR(10),
    MARKETING VARCHAR(10),
    HR VARCHAR(10),
    EDUCATION VARCHAR(10)
);


CREATE TABLE RESUME (
	RESUME_NUM INT(10) PRIMARY KEY,
    COMP_ID INT(10),
    STU_INTEREST VARCHAR(20),
    STU_WAGE INT(100),
    STU_EMP_TYPE VARCHAR(30),
    STU_CAREER VARCHAR(30),
    STU_CERTI VARCHAR(30),
    STU_INTRO VARCHAR(5000),
    STU_GRAD_YN CHAR(1),
    STU_AWARDS VARCHAR(20),
    STU_NUM INT(10),
    FOREIGN KEY(STU_NUM) REFERENCES STUDENT(STU_NUM),
    FOREIGN KEY(COMP_ID) REFERENCES JOB_FIELD(COMP_ID)
);


CREATE TABLE RESUME_BOARD (
	STU_POST_NUM INT(10) PRIMARY KEY,
    RESUME_NUM INT(10),
    COMP_ID INT(10),
    STU_POST_TITLE VARCHAR(99),
    STU_POST_DATE DATE,
    FOREIGN KEY(RESUME_NUM) REFERENCES RESUME(RESUME_NUM)
);

CREATE TABLE CAREER_DEPART (
	EMPLOYEE_NUM INT(100) PRIMARY KEY,
    EMPLOYEE_ID VARCHAR(100),
    EMPLOYEE_PW VARCHAR(100),
    EMPLOYEE_NAME VARCHAR(100),
    EMPLOYEE_EMAIL VARCHAR(100),
    EMPLOYEE_PHONE INT(100),
    EMPLOYEE_DEPARTMENT VARCHAR(10)
);

CREATE TABLE COMPANY (
	COMP_NO INT(100) PRIMARY KEY,
    COMP_ID INT(10),
    COMP_PW VARCHAR(100),
    COMP_NAME VARCHAR(40),
    COMP_PHONE VARCHAR(20),
    COMP_TYPE CHAR(10),
    COMP_HOMEPAGE VARCHAR(320),
    COMP_LOCATION VARCHAR(200),
    COMP_LOGO VARCHAR(200),
	FOREIGN KEY(COMP_ID) REFERENCES JOB_FIELD(COMP_ID)
);

CREATE TABLE JOB_INFO (
	JB_INFO_NUM INT(100) PRIMARY KEY,
    JB_POSITION VARCHAR(30),
    JB_EMP_TYPE VARCHAR(10),
    JB_SALARY VARCHAR(100),
    JB_CAREER VARCHAR(500),
    JB_EDU VARCHAR(100),
    JB_LOCATION VARCHAR(100),
    JB_JOB VARCHAR(500),
    JB_QUAL VARCHAR(500),
    JB_DOCS VARCHAR(500),
    JB_REGDATE DATE,
    JB_ENDDATE DATE,
    COMP_NO INT(10),
    JB_TIME VARCHAR(10),
	COMP_ID INT(10),
    FOREIGN KEY(COMP_NO) REFERENCES COMPANY(COMP_NO),
    FOREIGN KEY(COMP_ID) REFERENCES COMPANY(COMP_ID)
);


CREATE TABLE SCREENING_BOARD (
	AUTH_BOARD_NUM INT(100) PRIMARY KEY,
    EMPLOYEE_NUM INT(100),
    JB_POSITION VARCHAR(30),
    JB_INFO_NUM INT(100),
    JB_NAME VARCHAR(1000),
    AUTH_YN VARCHAR(10),
    FOREIGN KEY(EMPLOYEE_NUM) REFERENCES CAREER_DEPART(EMPLOYEE_NUM)
);


CREATE TABLE CAREER_DEPART (
	EMPLOYEE_NUM INT(100) PRIMARY KEY,
    EMPLOYEE_ID VARCHAR(30),
    EMPLOYEE_PW VARCHAR(30),
    EMPLOYEE_NAME VARCHAR(10),
    EMPLOYEE_EMAIL VARCHAR(40),
    EMPLOYEE_PHONE INT(20)
);

CREATE TABLE AUTHORIZED_BOARD (
	AUTH_NUM INT(10) PRIMARY KEY,
    AUTH_BOARD_NUM INT(100),
    COMP_POST_TITLE VARCHAR(10),
    FOREIGN KEY(AUTH_BOARD_NUM) REFERENCES SCREENING_BOARD(AUTH_BOARD_NUM)
);

CREATE TABLE DENIED_BOARD (
	DENIED_BOARD_NUM INT(10) PRIMARY KEY,
    AUTH_BOARD_NUM INT(10),
    FOREIGN KEY(AUTH_BOARD_NUM) REFERENCES SCREENING_BOARD(AUTH_BOARD_NUM)
);


CREATE TABLE JOB_MATCH (
	MATCH_NUM INT(10) PRIMARY KEY,
	AUTH_BOARD_NUM INT(10),
    RESUME_NUM INT(10),
    FOREIGN KEY(AUTH_BOARD_NUM) REFERENCES SCREENING_BOARD(AUTH_BOARD_NUM),
	FOREIGN KEY(RESUME_NUM) REFERENCES RESUME(RESUME_NUM)
);
